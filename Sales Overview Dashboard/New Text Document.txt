import pandas as pd
import random
from datetime import datetime, timedelta

# Generate sample data
random.seed(42)

rows = []
start_date = datetime(2025, 1, 1)

regions = ["North", "South", "East", "West"]
products = ["Laptop", "Phone", "Tablet", "Monitor", "Keyboard", "Mouse"]
categories = ["Electronics", "Accessories"]
segments = ["Consumer", "Corporate", "Home Office"]

for i in range(1, 201):
    order_date = start_date + timedelta(days=random.randint(0, 364))
    product = random.choice(products)
    category = "Electronics" if product in ["Laptop", "Phone", "Tablet", "Monitor"] else "Accessories"
    sales = round(random.uniform(50, 2000), 2)
    profit = round(sales * random.uniform(0.05, 0.3), 2)

    rows.append([
        f"ORD{i:04d}",
        order_date.strftime("%Y-%m-%d"),
        random.choice(regions),
        product,
        category,
        random.choice(segments),
        sales,
        profit
    ])

columns = [
    "OrderID",
    "OrderDate",
    "Region",
    "Product",
    "Category",
    "CustomerSegment",
    "Sales",
    "Profit"
]

df = pd.DataFrame(rows, columns=columns)

file_path = "/mnt/data/powerbi_sales_dataset_day1.csv"
df.to_csv(file_path, index=False)
