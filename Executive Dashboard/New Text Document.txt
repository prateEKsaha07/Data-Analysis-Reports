regions = ["North", "South", "East", "West"]
categories = ["Electronics", "Furniture", "Office Supplies"]

data = []

for date in dates:
    for region in regions:
        for category in categories:
            revenue = np.random.randint(5000, 20000)
            cost = revenue * np.random.uniform(0.55, 0.8)
            profit = revenue - cost
            data.append([date, region, category, revenue, round(cost, 2), round(profit, 2)])

df = pd.DataFrame(data, columns=["Date", "Region", "Category", "Revenue", "Cost", "Profit"])
