import pandas as pd
import random
from datetime import datetime, timedelta

random.seed(7)

rows = []
start_date = datetime(2025, 6, 1)

channels = ["Social Media", "Email", "Google Ads", "Affiliate", "Direct"]
campaigns = ["Summer Sale", "Brand Awareness", "Product Launch", "Retargeting"]
regions = ["North", "South", "East", "West"]

for i in range(1, 251):
    date = start_date + timedelta(days=random.randint(0, 119))
    impressions = random.randint(1000, 50000)
    clicks = random.randint(100, impressions // 5)
    conversions = random.randint(10, clicks // 2)
    cost = round(random.uniform(100, 5000), 2)
    revenue = round(conversions * random.uniform(20, 150), 2)

    rows.append([
        f"CMP{i:04d}",
        date.strftime("%Y-%m-%d"),
        random.choice(channels),
        random.choice(campaigns),
        random.choice(regions),
        impressions,
        clicks,
        conversions,
        cost,
        revenue
    ])

columns = [
    "CampaignID",
    "Date",
    "Channel",
    "CampaignType",
    "Region",
    "Impressions",
    "Clicks",
    "Conversions",
    "Cost",
    "Revenue"
]

df = pd.DataFrame(rows, columns=columns)

file_path = "/mnt/data/powerbi_marketing_dataset_day2.csv"
df.to_csv(file_path, index=False)
